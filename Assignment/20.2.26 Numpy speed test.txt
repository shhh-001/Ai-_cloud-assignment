import time
import numpy as np

size = 1000000

# Python List
start = time.time()
lst = list(range(size))
result1 = [x * 2 for x in lst]
end = time.time()
print("Python List Time:", end - start)


# NumPy Array
start = time.time()
arr = np.arange(size)
result2 = arr * 2
end = time.time()
print("NumPy Array Time:", end - start)
